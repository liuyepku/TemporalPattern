141 template <class CONTAINER> inline
142 istream&
143 stl_container_input
144  (
145  istream& is,
146  CONTAINER& a_container,
147  const char * const a_start = KEYWORD_STL_START,
148  const char * const a_inter = KEYWORD_STL_INTER,
149  const char * const a_final = KEYWORD_STL_FINAL
150  )
151  {
152  read_until(is,a_start);
153  while (!is_followed_by(is,a_final,false))
154  {
155  typename CONTAINER::value_type element;
156  is >> element;
157  a_container.insert(a_container.end(),element);
158  is_followed_by(is,a_inter,false); // remove if found
159  }
160  return is;
161  }
